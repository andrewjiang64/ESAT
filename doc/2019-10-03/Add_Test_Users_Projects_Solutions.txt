-- Insert UserMunicipalities
INSERT INTO public."UserMunicipalities"
	SELECT 
		row_number() over () as "Id", 
		U."Id" as "UserId", 
		M."Id" as "MunicipalityId" 
	FROM
		public."User" AS U 
		INNER JOIN 
		public."Municipality" AS M 
		ON U."UserName" != M."Name";

-- Insert UserWatersheds
INSERT INTO public."UserWatersheds"
	SELECT 
		row_number() over () as "Id", 
		U."Id" as "UserId", 
		W."Id" as "WatershedId" 
	FROM
		public."User" AS U 
		INNER JOIN 
		public."Watershed" AS W 
		ON U."UserName" != W."Name";

-- Insert UserParcels
INSERT INTO public."UserParcels"
	SELECT 
		row_number() over () as "Id", 
		U."Id" as "UserId", 
		P."Id" as "ParcelId" 
	FROM
		public."User" AS U

		public."Parcel" AS P
